const express = require("express");
const cors = require("cors");
const { v4: uuidv4 } = require("uuid");

const app = express();
app.use(cors());
app.use(express.json());

let providers = require("./providers.json");
let bookings = [];

function calculateScore(p, user) {
  return (
    (1 / p.distance_km) * 0.2 +
    p.rating * 0.15 +
    p.reliability * 0.15 +
    (1 - p.cancellation_rate) * 0.1 +
    (p.available ? 1 : 0) * 0.2 +
    (p.price_base <= user.budget ? 1 : 0) * 0.1
  );
}

app.post("/process", (req, res) => {
  const userInput = req.body.text;

  // MOCK INTENT EXTRACTION
  const intent = {
    service: "AC Repair",
    location: "G-13",
    time: "10:00 AM",
    budget: 1200
  };

  // ADD DISTANCE MOCK
  providers = providers.map(p => ({
    ...p,
    distance_km: Math.random() * 5,
    available: Math.random() > 0.3
  }));

  // RANK
  const ranked = providers
    .map(p => ({
      ...p,
      score: calculateScore(p, intent)
    }))
    .sort((a, b) => b.score - a.score);

  const best = ranked[0];

  const booking = {
    id: uuidv4(),
    provider: best.name,
    time: intent.time,
    status: "confirmed"
  };

  bookings.push(booking);

  res.json({
    intent,
    providers: ranked.slice(0, 3),
    selected: best,
    booking
  });
});

app.listen(5000, () => console.log("Server running on 5000"));

const express = require("express");
const cors = require("cors");
const { v4: uuidv4 } = require("uuid");

const app = express();
app.use(cors());
app.use(express.json());

let providers = require("../data/providers.json");
let bookings = [];

function scoreProvider(p, user) {
  return (
    (1 / (p.distance_km + 0.1)) * 0.2 +
    p.rating * 0.15 +
    p.reliability * 0.15 +
    (1 - p.cancellation_rate) * 0.1 +
    (p.available ? 1 : 0) * 0.2 +
    (p.price_base <= user.budget ? 1 : 0) * 0.1
  );
}

function mockIntent(text) {
  return {
    service: "AC Repair",
    location: "G-13",
    time: "Tomorrow 10:00 AM",
    budget: 1200,
    urgency: text.includes("urgent") ? "high" : "normal"
  };
}

app.post("/process", (req, res) => {
  const text = req.body.text;

  const intent = mockIntent(text);

  providers = providers.map(p => ({
    ...p,
    distance_km: Math.random() * 6,
    available: Math.random() > 0.3
  }));

  const ranked = providers
    .map(p => ({ ...p, score: scoreProvider(p, intent) }))
    .sort((a, b) => b.score - a.score);

  const selected = ranked[0];

  // ⚠️ simulate failure (20% chance)
  const providerCancelled = Math.random() < 0.2;

  let finalProvider = selected;

  if (providerCancelled) {
    finalProvider = ranked[1];
  }

  const booking = {
    id: uuidv4(),
    provider: finalProvider.name,
    status: "confirmed"
  };

  bookings.push(booking);

  res.json({
    intent,
    ranked: ranked.slice(0, 3),
    selected: finalProvider,
    fallback_used: providerCancelled,
    booking
  });
});

app.listen(5000, () => console.log("Backend running on 5000"));

import React, { useState } from "react";
import { View, Text, TextInput, Button, ScrollView } from "react-native";
import axios from "axios";

export default function App() {
  const [text, setText] = useState("");
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(false);

  const run = async () => {
    setLoading(true);
    const res = await axios.post("http://YOUR_IP:5000/process", {
      text
    });
    setData(res.data);
    setLoading(false);
  };

  return (
    <ScrollView style={{ padding: 20 }}>
      <Text style={{ fontSize: 20 }}>BazaarAI</Text>

      <TextInput
        placeholder="Enter request..."
        value={text}
        onChangeText={setText}
        style={{ borderWidth: 1, marginVertical: 10 }}
      />

      <Button title="Process Request" onPress={run} />

      {loading && <Text>🧠 Thinking... Agents working...</Text>}

      {data && (
        <View>
          <Text>📍 Service: {data.intent.service}</Text>
          <Text>📍 Location: {data.intent.location}</Text>

          <Text style={{ marginTop: 10 }}>🏆 Top Providers:</Text>

          {data.ranked.map((p, i) => (
            <Text key={i}>
              {i + 1}. {p.name} (Score: {p.score.toFixed(2)})
            </Text>
          ))}

          <Text style={{ marginTop: 10 }}>
            ✅ Selected: {data.selected.name}
          </Text>

          {data.fallback_used && (
            <Text>⚠️ Fallback activated (provider cancelled)</Text>
          )}

          <Text>📦 Booking ID: {data.booking.id}</Text>
        </View>
      )}
    </ScrollView>
  );
}
